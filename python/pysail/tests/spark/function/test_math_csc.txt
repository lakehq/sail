>>> from pyspark.sql import functions as sf
>>> import math

>>> df = spark.range(1)
>>> df.select(sf.csc(sf.lit(math.radians(90)))).show()
+-----------------------+
|CSC(1.5707963267948966)|
+-----------------------+
|                    1.0|
+-----------------------+
>>> df.select(sf.csc(sf.lit(90))).show()
+------------------+
|           CSC(90)|
+------------------+
|1.1185724071637084|
+------------------+
>>> df.select(sf.csc(sf.lit(90.0))).show()
+------------------+
|         CSC(90.0)|
+------------------+
|1.1185724071637084|
+------------------+
>>> df.select(sf.csc(sf.lit(-90.0))).show()
+-------------------+
|         CSC(-90.0)|
+-------------------+
|-1.1185724071637084|
+-------------------+

>>> df.select(sf.csc(sf.lit(0))).show()
+------+
|CSC(0)|
+------+
|   inf|
+------+
>>> # Infinity in pyspark
