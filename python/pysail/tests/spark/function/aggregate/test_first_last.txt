>>> from pyspark.sql import functions as F
>>> from pyspark.sql import Window

>>> df = spark.createDataFrame(
...     [("Alice", 2), ("Bob", 5), ("Alice", None)],
...     ("name", "age"),
... )
>>> result = (
...     df.orderBy("age")
...     .select("name", "age")
...     .groupBy("name")
...     .agg(F.first("age"))
... )
>>> sorted(result.collect(), key=lambda r: r.name)
[Row(name='Alice', first(age)=None), Row(name='Bob', first(age)=5)]

>>> result = (
...     df.orderBy("age")
...     .select("name", "age")
...     .groupBy("name")
...     .agg(F.last("age"))
... )
>>> sorted(result.collect(), key=lambda r: r.name)
[Row(name='Alice', last(age)=2), Row(name='Bob', last(age)=5)]

>>> df = spark.createDataFrame([(30, 2, 2), (10, 5, -20), (20, 1, 1)], ("v", "a", "b"))
>>> df.orderBy(df.a + df.b).select("v", "a", "b").groupBy().agg(F.first("v")).collect()
[Row(first(v)=10)]

>>> spark.sql("""WITH sorted AS (SELECT * FROM VALUES (10,'b'),(20,'a'),(30,'d'),(40,'c') AS t(v, k) ORDER BY k) SELECT first(v) AS result FROM sorted""").collect()
[Row(result=20)]

>>> df = spark.createDataFrame([(30, 'c'), (10, 'a'), (20, 'b')], ("v", "k"))
>>> df.orderBy("k").groupBy().agg(F.collect_list("v")).collect()
[Row(collect_list(v)=[10, 20, 30])]

>>> df = spark.createDataFrame([("Alice", 2), ("Bob", 5), ("Alice", None)], ("name", "age"))
>>> w = Window.partitionBy("name")
>>> result = df.orderBy("age").select("name", "age", F.first("age").over(w).alias("f"))
>>> sorted(result.collect(), key=lambda r: (r.name, r.age if r.age is not None else -1))
[Row(name='Alice', age=None, f=None), Row(name='Alice', age=2, f=None), Row(name='Bob', age=5, f=5)]

>>> df = spark.createDataFrame([("Alice", 2), ("Bob", 5), ("Alice", None)], ("name", "age"))
>>> w = Window.partitionBy("name")
>>> result = df.orderBy("age").select("name", "age", F.last("age").over(w).alias("l"))
>>> sorted(result.collect(), key=lambda r: (r.name, r.age if r.age is not None else -1))
[Row(name='Alice', age=None, l=2), Row(name='Alice', age=2, l=2), Row(name='Bob', age=5, l=5)]

>>> df = spark.createDataFrame([(1, 'a'), (2, 'a'), (3, 'a')], ("v", "k"))
>>> df.orderBy("k").groupBy().agg(F.first("v")).collect()
[Row(first(v)=1)]

>>> a = spark.createDataFrame([(2, 'b'), (1, 'a')], ("v", "k"))
>>> b = spark.createDataFrame([(4, 'd'), (3, 'c')], ("v", "k"))
>>> a.union(b).orderBy("k").groupBy().agg(F.first("v"), F.last("v")).collect()
[Row(first(v)=1, last(v)=4)]

>>> left = spark.createDataFrame([(2, 'b'), (1, 'a')], ("v", "k"))
>>> right = spark.createDataFrame([('a',), ('b',)], ("k",))
>>> left.join(right, "k").orderBy(F.desc("k")).groupBy().agg(F.first("v")).collect()
[Row(first(v)=2)]

>>> df = spark.createDataFrame([("Alice", 2), ("Bob", 5), ("Alice", None)], ("name", "age"))
>>> w = Window.partitionBy("name")

>>> result = df.orderBy("age").select("name", "age", F.first("age").over(w).alias("f"), F.last("age").over(w).alias("l"))
>>> sorted(result.collect(), key=lambda r: (r.name, r.age if r.age is not None else -1))
[Row(name='Alice', age=None, f=None, l=2), Row(name='Alice', age=2, f=None, l=2), Row(name='Bob', age=5, f=5, l=5)]

>>> result = df.orderBy("age").select("name", "age", (F.last("age").over(w) + F.last("age").over(w)).alias("s"))
>>> sorted(result.collect(), key=lambda r: (r.name, r.age if r.age is not None else -1))
[Row(name='Alice', age=None, s=4), Row(name='Alice', age=2, s=4), Row(name='Bob', age=5, s=10)]

>>> result = df.orderBy("age").select("name", "age", F.sum("age").over(w).alias("sm"), F.last("age").over(w).alias("l"))
>>> sorted(result.collect(), key=lambda r: (r.name, r.age if r.age is not None else -1))
[Row(name='Alice', age=None, sm=2, l=2), Row(name='Alice', age=2, sm=2, l=2), Row(name='Bob', age=5, sm=5, l=5)]

>>> spark.sql("SELECT listagg(k) AS s FROM (SELECT * FROM VALUES (30,'c'),(10,'a'),(20,'b') AS t(v, k) ORDER BY v)").collect()

>>> spark.sql("SELECT g, first(v) AS f FROM (SELECT * FROM VALUES (30, 1), (10, 1), (20, 2) AS t(v, g) ORDER BY v) GROUP BY g HAVING first(v) > 15").collect()
[Row(g=2, f=20)]

>>> spark.sql("SELECT g, first(v) AS f FROM (SELECT * FROM VALUES (30, 1), (10, 1), (20, 2) AS t(v, g) ORDER BY v) GROUP BY CUBE(g) ORDER BY g NULLS FIRST").collect()
