>>> import pyarrow as pa
>>> import pyarrow.compute as pc
>>> from pyspark.sql.functions import arrow_udf

>>> @arrow_udf("long")
... def sq(v: pa.Array) -> pa.Array:
...     return pc.multiply(v, v)
...
>>> spark.range(3).select(sq("id")).show()
+------+
|sq(id)|
+------+
|     0|
|     1|
|     4|
+------+

>>> from typing import Iterator

>>> @arrow_udf("long")

...         yield pc.add(v, 1)
...
