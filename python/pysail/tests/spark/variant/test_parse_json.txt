>>> spark.sql("""SELECT parse_json('{"name":"sail"}') AS result""").show()
+---------------+
|         result|
+---------------+
|{"name":"sail"}|
+---------------+

>>> spark.sql("""SELECT parse_json('{"level1": {"level2": {"level3": "value"}}}') AS result""").show(truncate=False)
+----------------------------------------+
|result                                  |
+----------------------------------------+
|{"level1":{"level2":{"level3":"value"}}}|
+----------------------------------------+

>>> spark.sql("""SELECT parse_json('{"user": {"profile": {"settings": {"theme": "dark", "notifications": true}}, "name": "Alice"}, "status": "active"}') AS result""").show(truncate=False)
+--------------------------------------------------------------------------------------------------------+
|result                                                                                                  |
+--------------------------------------------------------------------------------------------------------+
|{"status":"active","user":{"name":"Alice","profile":{"settings":{"notifications":true,"theme":"dark"}}}}|
+--------------------------------------------------------------------------------------------------------+

>>> spark.sql("""SELECT parse_json('{"company": {"departments": [{"name": "Engineering", "team": {"lead": "Bob", "members": 15}}, {"name": "Sales", "team": {"lead": "Carol", "members": 8}}], "founded": 2020}}') AS result""").show(truncate=False)
+-----------------------------------------------------------------------------------------------------------------------------------------------------------+
|result                                                                                                                                                     |
+-----------------------------------------------------------------------------------------------------------------------------------------------------------+
|{"company":{"departments":[{"name":"Engineering","team":{"lead":"Bob","members":15}},{"name":"Sales","team":{"lead":"Carol","members":8}}],"founded":2020}}|
+-----------------------------------------------------------------------------------------------------------------------------------------------------------+
